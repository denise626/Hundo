//get the values from the Page
//start or controller function
function getValues() {
    //get values from the page
    let startValue = document.getElementById("startValue").value;
    let endValue = document.getElementById("endValue").value;
  
    //We need to validate our input
    //parse into Integers
    startValue = parseInt(startValue);
    endValue = parseInt(endValue);
    
    if (Number.isInteger(startValue) && Number.isInteger(endValue)){
        
        //we call generateNumbers
        let numbers = generateNumbers(startValue, endValue);
        //we call displayNumbers
        displayNumbers(numbers);
    
    }else {
        alert("You must enter integers");
    }    
    
}

//generate numbers from startValue to endValue
//logic function(s)
function generateNumbers(sValue, eValue) {

        let numbers = [];

        //we want to get all numbers from start to end

        for(let index = sValue; index <= eValue; index++){

            //this will execute in a loop until index = eValue
            numbers.push(index);
        }
        return numbers;

}

//display the numbers and mark even numbers bold
//display or view functions
function displayNumbers(numbers) {

    let templateRows ="";

    for (let index = 0; index < numbers.length; index++) {
        
        let className = "even";

        let number = numbers[index];

        if(number % 2 === 0) {
            className = "even";
        }
        else{
            className = "odd";

        }
        //This does render correctly with Prism see the source 
        templateRows += `${number}`;
        
    }

    document.getElementById("results").innerHTML = templateRows;
}

The Code for Hundo is structured in three functions.

Hundo Functions

The function getValues allows us to get the starting and ending values that we entered into the empty value boxes. By using parseInt, we make sure that our numbers are read as integers and not strings. The if Number.isInteger is used to determine if the numbers are numbers or text values. By using &&, both the startValue and endValue have to be entered as numbers for the statement to be true and results will be displayed. If both are not entered in as numbers, the statement will revert to false and results will not display. The else statment is triggered with an alert to enter integers.

The function generateNumbers allows us to get all numbers from the startValue to the endValue. The for loop allows us to get all the numbers from the startValue to the endValue and in between by using index++. Index++ allows a number to be added each time to the loop until it reaches the last number that was to be entered as the endValue and are returned using the return keyword.

The function displayNumbers allows us to use a for loop to display all the numbers in the results table. An if and else statement is in the for loop to determine if the number is even or odd. The even numbers will be displayed in bold which is written in the css stylesheet.
